```

│   ├── redis -- 缓存

│   │   └── video

│   ├── upload

    │   ├── handlers

    │   │   ├── user -- 请求
    │   │   └── video

    │   └── rpc  -- 调用 rpc
    │       ├── mock
    │       │   └── user

    │   ├── pack

    │   │   ├── chat

    │   │   └── user

        ├── model -- 数据库
        │   └── db
        ├── pack

        └── tmp
```
